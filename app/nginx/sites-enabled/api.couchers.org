server {
    listen 443 http2 ssl default_server;

    server_name api-test.coucher.org;

    access_log /var/log/nginx/api.couchers.org.access.log;
    error_log /var/log/nginx/api.couchers.org.error.log;

    ssl_certificate /certs/live/api-test.coucher.org/fullchain.pem;
    ssl_certificate_key /certs/live/api-test.coucher.org/privkey.pem;

    location /org.couchers. {
        proxy_http_version 1.1;
        proxy_pass http://envoy:8888;
    }

    location = / {
        add_header Content-Type text/plain;

        set $msg "This is the API for the Couchers.org app. If you're ";
        set $msg "interested in helping out in building the next-generation ";
        set $msg "couch-surfing platform with us, or if you're just ";
        set $msg "interested in API access, please email us.";

        return 200 $msg;
    }

    location / {
        return 404;
    }
}
